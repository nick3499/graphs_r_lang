top #!/usr/bin/env R X <- c(1,5.5,10,14.5,19,23.5,28,32.5) Y <- c(2,6.5,11,15.5,20,24.5,29,33.5) k <- 8 min <- 2 max <- 23 name <- "Radon" plot(X, Y, type="o", col="581", xlab = paste(name, "'s Profits", sep = ""), ylab = paste("Points from ", min, " to ", max, sep = "")) title(main = paste("Sector ", k, " for ", name, sep = ""), font.main = 2, col.main = "#CC6600")
